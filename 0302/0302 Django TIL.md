##### views.py

```python
def dtl_practice(request):
    name = 'Kim ae-ri'
    foods = ['짬뽕', '치킨', '막국수', '냉면', '샤브샤브']
    users = []
    info = {
        'name' : name,
    }
    context = {
        'info' : info,
        'foods' : foods,
        'users' : users,
        'color' : 'red',
        'colors' : 'blue',
    }
    return render(request, 'dtl_practice.html', context)
```
